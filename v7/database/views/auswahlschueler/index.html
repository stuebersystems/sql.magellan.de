    Abfrage

    SQL 
     SELECT                                  
  T.*                                      
FROM                                       
  "tblAuswahlSchueler" T                   
LEFT JOIN                                  
  "tblBenutzer" B                          
ON                                         
  B."Mandant" = T."Mandant" AND            
  B."Kennung" = T."Benutzer"               
WHERE                                      
  B."Kennung" = USER
